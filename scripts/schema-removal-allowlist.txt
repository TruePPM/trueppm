# Intentionally-removed OpenAPI paths / schemas, declared so the
# schema-regression-vs-main guard (scripts/check-schema-regression.py) does not
# treat a deliberate, reviewed removal as an accidental drop.
#
# One entry per line. A bare line is a path; "schema:<Name>" is a component
# schema. "#" starts a comment. Entries go stale harmlessly once merged (the
# baseline no longer contains them) and can be pruned in any later cleanup.

# #846 — dead "ghost" suggestion routes: detail=True @action auto-registered
# these parameterless variants, which always 404 (suggestion_pk never bound).
# The real routes (tasks/{pk}/suggestions/{suggestion_pk}/{verb}/) are unchanged.
/api/v1/tasks/{id}/suggestions/accept/
/api/v1/tasks/{id}/suggestions/decline/
/api/v1/tasks/{id}/suggestions/revoke/

# #897 — httpOnly refresh-cookie migration: the token-refresh endpoint no longer
# accepts/returns the refresh token in the request/response body (it is read
# from, and rotated into, an httpOnly Secure SameSite=Strict cookie). The
# SimpleJWT TokenRefreshView (which generated these body schemas) was replaced
# by CookieTokenRefreshView. The /api/v1/auth/token/refresh/ path is unchanged.
schema:TokenRefresh
schema:TokenRefreshRequest

# #997 — login response no longer carries the refresh token: CookieTokenObtainPairView
# delivers refresh as an httpOnly cookie and returns only `access` in the body
# (the #897 cookie migration). An explicit @extend_schema replaces SimpleJWT's
# inherited TokenObtainPair response, so that phantom schema is no longer emitted.
# The /api/v1/auth/token/ path and its TokenObtainPairRequest body are unchanged.
schema:TokenObtainPair

# #1017 — signal-privacy action URLs renamed snake_case → kebab-case to match the
# other 47 action paths (raise-ceiling / ratchet-down replace these). Signal
# privacy is new in 0.3 with no shipped consumers, so this is a clean rename with
# no redirect shim; the new paths are present, only the old segments are removed.
/api/v1/projects/{project_pk}/signal-privacy/raise_ceiling/
/api/v1/projects/{project_pk}/signal-privacy/ratchet_down/

# #695 (ADR-0210) — dead-letter write-action rename: the legacy retry/dismiss
# actions (raw send_task side channel + note-less dismiss) are replaced by
# requeue (durable workflow-backend round-trip + operator backoff) and drop
# (soft-remove + audit note), plus the new bulk requeue_all/drop_all. The old
# actions had no shipped consumers (#694 was read-only); the new paths are
# present, only the old segments are removed.
/api/v1/admin/failed-tasks/{id}/retry/
/api/v1/admin/failed-tasks/{id}/dismiss/

# #2108 (ADR-0517) — SSO re-platform onto django-allauth socialaccount: the
# singleton /workspace/sso/ config endpoint is replaced by the multi-provider
# /workspace/sso/providers/ collection (list/create + {slug}/ item +
# {slug}/test-connection/), and the OIDCProvider* / OIDCTestConnection* /
# DefaultRoleEnum serializers are superseded by the SsoProvider* serializers. SSO
# is unshipped (0.4) with no external consumers; the new paths/schemas are present,
# only the old singleton surface is removed.
/api/v1/workspace/sso/
/api/v1/workspace/sso/test-connection/
schema:OIDCDiscoverResponse
schema:OIDCProviderRead
schema:OIDCProviderWriteRequest
schema:OIDCTestConnectionRequestRequest
schema:OIDCTestConnectionResponse
schema:DefaultRoleEnum
